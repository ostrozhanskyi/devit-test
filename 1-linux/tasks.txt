#1 Вывести список всех файлов в папке "html"
ls -la /var/www/html #Простая команда для просмотра содержимого директории



#2 Найти в файле "access.log" все упоминания строки "404"
grep 404 /var/logs/access.log #Для поиска необходимо указать только строку
#для поиска и путь к файлу



#3 На основании файла "404.html" сделать файл "403.html" заменив в его
#содержимом все упоминания "404" на "403"
cp /var/www/html/404.html /var/www/html/403.html #Копирую файл
sed -i 's/404/403/g' /var/www/html/403.html #Глобальная замена и сохранение



#4 Модифицировать права доступа и владельца файла "index.html" следующим
#образом: текущий пользователь должен иметь права на чтение/запись, группа 
#"www-data" на чтение, все остальные не имеют доступа к файлу
chmod 640 /var/www/html/index.html #Изменение прав для владельца - 4+2 - чтение
#и запись, для группы - 4 - только чтение, для других - 0 - нет прав доступа
chgrp www-data /var/www/html/index.html #Изменение группы, если это не было
#сделано ранее. Возможно необходимы будут повышенные привилегии для выполнения
#команды



#5 Сделать символическую ссылку на файл access.log в папке html
ln -s /var/log/access.log /var/www/html #При необходимости можно указать другое
имя для ссылки. Также можно создать hardlink не указывая ключ "-s"



#6 Вывести на экран все файлы в папке "var"
find /var -type f #При использовании "ls" показываються и директории, а при 
#использовании "-type f" в find поиск осуществляется только по файлам



#7 Создать новый файл "503.html" в папке "html" с содержимым
echo "<html>
  <head>
    <title>503</title>
  </head>
</html>" > 503.html
#Текст записывается в новый файл, но если файл с таким именем уже существовал
#он будет перезаписан
